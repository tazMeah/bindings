index.html

    <!DOCTYPE html>
    <html lang="en" ng-app="myApp">
        <head>
            <meta charset="UTF-8">
            <meta name="viewport" content="width=device-width, initial-scale=1.0">
            <meta http-equiv="X-UA-Compatible" content="ie=edge">
            <title>Document</title>
        </head>
        <body>
            <outer></outer>

            <script src="https://code.angularjs.org/1.6.5/angular.min.js"></script>
            <script src="module.js"></script>
            <script src="outer.js"></script>
            <script src="inner.js"></script>
        </body>
    </html>

module.js

    "use strict";

    var app = angular.module("myApp", [])

outer.js

    "use strict";

    const outer = {
        template: `
        <inner outer-f="$ctrl.outerF();"></inner>
        `,
        controller: function() {
            const vm = this;
            vm.outerF = function(){
                console.log("The method is in the outer component. The button is in the inner component.");
            }
            
        }
    
    };
    
    app.component("outer", outer)

inner.js

    "use strict";

    const inner = {
        bindings: {
            outerF: "&"
        },
        template: `
        <button type="button" ng-click="$ctrl.outerF()">Button in Inner Component</button>
        `
        
    
    };
    
    app.component("inner", inner)
